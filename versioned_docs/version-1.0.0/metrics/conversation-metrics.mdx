```
IMConversationTimeSeriesMetricsQueryBuilder builder = (IMConversationTimeSeriesMetricsQueryBuilder) new IMConversationTimeSeriesMetricsQueryBuilder()
        .startDate(new TimeStamp(1000876164181L))                // Mandatory
        .endDate(new Timestamp(100003765471634L))                // Mandatory
        .granularity(Granularity.DAY.name());                    // Mandatory

builder.channelIds(Arrays.asList(100000076511171L));                    // Optional
builder.integrationService(IMIntegrationServices.WHATSAPP);                // Optional
builder.dimensionAndValues(TimeSeriesDimensions.CONVERSATION_STATUS.name(), new ArrayList&lt;&gt;());    // Optional
builder.basedOn(BasedOnFilters.END_TIME.name());                    // Optional
builder.agentIds(Arrays.asList(10000016271517L));                    // Optional

IMTimeSeriesMetrics metrics = ZohoIM.getInstance(IM_ORG_ID, SERVICE_ORG_ID)
        .getIMTimeSeriesMetricsAPI(USER_ID)
        .getConversationTimeSeriesMetrics(builder);
```
