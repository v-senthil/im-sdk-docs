        .startDate(new Timestamp(1000876164181L))            // Mandatory
        .endDate(new Timestamp(100003765471634L))            // Mandatory
        .integrationService(IMIntegrationServices.WHATSAPP);
builder.channelIds(Arrays.asList(100000076511171L));            // Optional
IMMetrics metrics = ZohoIM.getInstance(IM_ORG_ID, SERVICE_ORG_ID)
        .getIMMetricsAPI(USER_ID)
        .getAgentResponsivenessMetrics(builder);
